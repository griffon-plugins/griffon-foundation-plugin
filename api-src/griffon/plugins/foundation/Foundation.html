310     private static final String ERROR_DESCRIPTION_BLANK = "Argument 'description' must not be blank";
311 
312     public static Foundation findByDescription(@Nonnull String description) {
313         requireNonBlank(description, "Icon description must not be blank.");
314         String[] parts = description.split(":");
315         for (Foundation font : values()) {
316             if (font.getDescription().equals(parts[0])) {
317                 return font;
318             }
319         }
320         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
321     }

323     private String description;
324     private char code;
325 
326     Foundation(@Nonnull String description, char code) {
327         this.description = description;
328         this.code = code;
329     }

331     @Nonnull
332     public String getDescription() {
333         return description;
334     }
335

337         return code;
338     }

340     @Nonnull
341     public static IllegalArgumentException invalidDescription(@Nonnull String description) {
342         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
343         throw new IllegalArgumentException("Description " + description + " is not a valid Foundation icon description");
344     }

346     @Nonnull
